    @Override
    public void close() {
        if (!isAlive()) return;

        try {
            logger.debug("Close db: {}", name);
            db.close();

            alive = false;
        } catch (IOException e) {
            logger.error("Failed to find the db file on the close: {} ", name);
        }
    }
